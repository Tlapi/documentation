---
title: 'Find contact'
api: 'POST https://api2.ecomailapp.cz/search'
description: 'Search for a subscriber globally across the account by email address.'
---

<ParamField body="query" type="string" required>
  Must be a valid email address.
</ParamField>

<RequestExample>
```bash cURL
curl -X POST https://api2.ecomailapp.cz/search \
  -H "key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "query": "foo@bar.cz"
  }'
```

```php PHP
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, 'https://api2.ecomailapp.cz/search');
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'key: YOUR_API_KEY',
    'Content-Type: application/json',
]);
curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode([
    'query' => 'foo@bar.cz',
]));
$response = curl_exec($ch);
curl_close($ch);
```
</RequestExample>
