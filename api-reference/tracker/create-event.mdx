---
title: 'Create an event'
api: 'POST https://api2.ecomailapp.cz/tracker/events'
description: 'Track a custom event for a subscriber. Supplements the JavaScript tracker.'
---

<ParamField body="event" type="object" required>
  Event details.

  <Expandable title="event properties">
    <ParamField body="event.email" type="string" required>
      Subscriber's email address.
    </ParamField>

    <ParamField body="event.category" type="string">
      Event category.
    </ParamField>

    <ParamField body="event.action" type="string">
      Event action.
    </ParamField>

    <ParamField body="event.label" type="string">
      Event label.
    </ParamField>

    <ParamField body="event.property" type="string">
      Event property.
    </ParamField>

    <ParamField body="event.value" type="string">
      Event value.
    </ParamField>
  </Expandable>
</ParamField>

<RequestExample>
```bash cURL
curl -X POST https://api2.ecomailapp.cz/tracker/events \
  -H "key: YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "event": {
      "email": "foo@bar.cz",
      "category": "category",
      "action": "action",
      "label": "label",
      "property": "property",
      "value": "value"
    }
  }'
```

```php PHP
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, 'https://api2.ecomailapp.cz/tracker/events');
curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
curl_setopt($ch, CURLOPT_POST, true);
curl_setopt($ch, CURLOPT_HTTPHEADER, [
    'key: YOUR_API_KEY',
    'Content-Type: application/json',
]);
curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode([
    'event' => [
        'email' => 'foo@bar.cz',
        'category' => 'category',
        'action' => 'action',
        'label' => 'label',
        'property' => 'property',
        'value' => 'value',
    ],
]));
$response = curl_exec($ch);
curl_close($ch);
```
</RequestExample>
